(function () {
        "use strict";

        var utils = (function() {
            var boundingPrefs = {CENTER: 0, BOTTOM_CENTER: 1};
            
            function displayElementText(centerX, centerY, id) {
                var titleBoxContainer = document.getElementsByClassName("title-box")[0];
                var titleBox = document.createElement("title-box");
                titleBox.className = "title-box";
                titleBox.innerHTML = labels[id];
                titleBoxContainer.appendChild(titleBox);
                styleElement(titleBox, (centerX - titleBox.offsetWidth / 2), (centerY + titleBox.offsetHeight));
            }
            
            function styleElement(element, left, top) {
                element.style.left = left + "px";
                element.style.top = top + "px";
            }
            
            function getBoundingCenter(minX, maxX, minY, maxY, pref) {
                var x = parseInt((minX + maxX) / 2, 10);
                var y;
                if (pref === boundingPrefs.CENTER) {
                    y = parseInt((minY + maxY) / 2, 10);
                } else {
                    y = parseInt(maxY + 5, 10);
                }
                return [x, y];
            }
            
            function getAreaCenter(coords, pref) {
                var coordsArray = coords.split(","),
                    center = [];
                var coord, maxX, maxY, minX = maxX = parseInt(coordsArray[0], 10),
                    minY = maxY = parseInt(coordsArray[1], 10);
                for (var i = 0; i < coordsArray.length; i++) {
                    coord = parseInt(coordsArray[i], 10);
                    if (i % 2 === 0) {
                        if (coord < minX) {
                            minX = coord;
                        } else if (coord > maxX) {
                            maxX = coord;
                        }
                    } else {
                        if (coord < minY) {
                            minY = coord;
                        } else if (coord > maxY) {
                            maxY = coord;
                        }
                    }
                }
                return getBoundingCenter(minX, maxX, minY, maxY, pref);
            }
  
            return {boundingPrefs: boundingPrefs,
                    displayElementText: displayElementText,
                    getAreaCenter: getAreaCenter};
        })();
        
        var controller = (function() {
        
            document.getElementById("map-container").onload = function () {
                var areas = document.querySelectorAll("area");
                for (var i = 0; i < areas.length; i++) {
                    var centroid = utils.getAreaCenter(areas[i].getAttribute("coords"), utils.boundingPrefs.BOTTOM_CENTER);
                    utils.displayElementText(centroid[0], centroid[1], i);
                }
            };
            
            return { };
        })();
